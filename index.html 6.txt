    <script>
        function showPage(pageId) {
            document.querySelectorAll('main').forEach(m => m.classList.add('hidden-page'));
            const target = document.getElementById(pageId);
            target.classList.remove('hidden-page');
            target.classList.add('page-transition');
            window.scrollTo(0, 0);
        }
        const textElement = document.getElementById('dynamic-text');
        const texts = ["Selamat Datang di Sekolah Budi Bahasa", "Membangun Masa Depan Berkarakter", "Cerdas, Berbudaya, dan Beretika"];
        let index = 0;
        setInterval(() => {
            textElement.classList.add('fade-out');
            setTimeout(() => {
                index = (index + 1) % texts.length;
                textElement.innerText = texts[index];
                textElement.classList.remove('fade-out');
            }, 600);
        }, 4000);
    </script>
